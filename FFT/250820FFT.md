const int FFTN = 2.1e6;

struct complex {
    double r = 0, i = 0;
    complex(double _r, double _i) : r(_r), i(_i) {};
    complex() : r(0), i(0) {};
    inline complex operator+ (const complex o) const {
        return complex(r + o.r, i + o.i);
    }
    inline complex operator- (const complex o) const {
        return complex(r - o.r, i - o.i);
    }
    inline complex operator* (const complex o) const {
        return complex(r * o.r - i * o.i, r * o.i + o.r * i);
    }
};

void FFT_sort(auto a, int sz) {
    static int *rev = (int*)malloc(sizeof(int) * FFTN);
    int hsz = sz >> 1;
    rev[0] = 0;
    for (int i = 1; i < sz; i++) {
        rev[i] = rev[i >> 1] >> 1;
        if (i & 1) rev[i] |= hsz;
        if (i < rev[i]) {
            std::swap(a[i], a[rev[i]]);
        }
    }
}

void FFT(auto a, int sz, double opt) {
    FFT_sort(a, sz);
    for (int k = 2; k <= sz; k <<= 1) {
        complex dw = complex(std::cos(2.0 * M_PI / k), opt * std::sin(2.0 * M_PI / k));
        int h = k >> 1;
        for (int j = 0; j < sz; j += k) {
            complex w = complex(1, 0);
            for (int i = j; i < j + h; i++, w = w * dw) {
                complex tmp = w * a[h + i];
                a[i + h] = a[i] - tmp;
                a[i] = a[i] + tmp;
            }
        }
    }
    if (opt == -1.0) {
        for (int i = 0; i < sz; i++) {
            a[i].r /= sz;
            a[i].i /= sz;
        }
    }
}
